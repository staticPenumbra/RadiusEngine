 17 /**
 18  * @class
 19  * Class modeling a Javascript Menu
 20  *
 21  * @description
 22  * The Menu system for the Javascript application engine
 23  **/
 24  /**
 25  * @constructor
 26  */
 27 var ApplicationMenu = function(Origin, Dimensions, MenuItems, Title, Font) {
 28 	this.MenuOptions = MenuItems;//Optional element so add with a function
 29 	this.OptionCoordinates = new Array(); //location of each Menu Item
 30 	this.Background = null;//Optional element so add with a function
 31 	this.CursorImage = null;//Optional element so add with a function
 32 	this.BGMusic = null; //Optional element so add with a function
 33 	this.TextFont = Font;//Optional element so add with a function
 34 	this.TextSize = 10;//Optional element so add with a function
 35 	this.Animation = null;//Optional element so add with a function
 36 	this.BackgroundAnimation = null; //Optional element so add with a function
 37 	this.CursorAnimation = null; //Optional element so add with a function
 38 	this.MenuDimensions = Dimensions; //Menu Dimensions
 39 	this.Orig = Origin; //Upper left x coordinate
 40 	this.MainTitle = Title; //Menu Title
 41 	this.SelectedItem = 0; //Currently selected Menu Item
 42 	this.type = null; //Shape of the Menu
 43 	this.HUD = null; //Variable to indicate if the menu is a persistant HUD
 44 	this.Spacing = 80;
 45 }
 46 
 47 //--------------------------------------------------------------------------GET ACCESSORS---------------------------------------------
 48 /**
 49 * Returns the selected menu item
 50 * @return {Integer} The currently selected menu item
 51 */
 52 ApplicationMenu.prototype.GetCursorPosition = function(){
 53 	return(this.SelectedItem);
 54 }
 55 //--------------------------------------------------------------------------SET ACCESSORS---------------------------------------------
 56 /**
 57 * Sets the Cursor Image for the Menu
 58 * @param {String} CursorFile filepath of the cursor Image
 59 */
 60 ApplicationMenu.prototype.SetCursorImage = function(CursorFile){
 61 	if(CursorFile != null){
 62 		this.CursorImage = new Image();
 63 		this.CursorImage.src = CursorFile;
 64 	}
 65 }
 66 /**
 67 * Sets the font properties of the Menu
 68 * @param {String} FontName The name of the system font to set
 69 * @param {Integer} Size The text size 
 70 */
 71 ApplicationMenu.prototype.SetTextFont = function(FontName, Size){
 72 	if(FontName != null && Size != null){
 73 		this.TextFont = FontName;
 74 		this.TextSize = Size;
 75 	}
 76 }
 77 /**
 78 * Function to set the Menu background
 79 * @param {String} BackgroundFile File path to the background file
 80 */
 81 ApplicationMenu.prototype.SetBackground = function(BackgroundFile){
 82 	if(BackgroundFile != null){
 83 		if(BackgroundFile instanceof Image){
 84 			this.Background = BackgroundFile;
 85 		}else{
 86 			this.Background = new Image();
 87 			this.Background.src = BackgroundFile;
 88 		}
 89 	}
 90 }
 91 /**
 92 * Set the menu shape eg Circle, Square (Currently only supporting square)
 93 * @param {String} Shape Text string indicating the shape of the menu
 94 */
 95 ApplicationMenu.prototype.SetShape = function(Shape){
 96 	if(Shape != null && Shape == "Square"){
 97 	this.type = Shape;
 98 	}
 99 }
100 /**
101 * Change Selected Menu Item
102 * @param {Integer} MenuOption Array element number of the menu selection
103 */
104 ApplicationMenu.prototype.ChangeCursorPosition = function(MenuOption) {
105 	if(MenuOption != null){
106 	this.SelectedItem = MenuOption;
107 	}
108 }

115 /**
116 * Function to display the menu on screen
117 * @param {CanvasContext} ctx Handle to the current canvas context
118 */
119 ApplicationMenu.prototype.Display = function(ctx) {
120 //In order to display a menu in the current canvas element we need:
121 // A canvas, A position, menu dimensions, A border, A background, A font, menu entries, a cursor image, an input
122 	//Origin, Dimensions, MenuItems, Title, Font, Canvas
123 	if(ctx != null && this.Orig != null && this.MenuOptions != null && this.MainTitle != null){
124 		switch(this.type){
125 			case "Square":
126 			//Draw Background Layer
127 			ctx.drawImage(this.Background, this.Orig[0], this.Orig[1]);
128 			//Draw Text Layer
129 			ctx.font = this.TextFont;
130 			for(var i = 0; i < this.MenuOptions.length; i++){
131 				var x = this.Spacing * 2 + this.Orig[0];
132 				var y = this.Spacing + i*this.Spacing + this.Orig[1];
133 				ctx.fillText(this.MenuOptions[i], x, y);
134 				this.OptionCoordinates.push(new Array(x, y));
135 			}
136 			if(this.CursorImage != null){
137 				//Place the cursor next to the currently selected Item
138 				ctx.drawImage(this.CursorImage, this.OptionCoordinates[this.SelectedItem][0] - 50, this.OptionCoordinates[this.SelectedItem][1] - 10);
139 				//this.ctx.drawImage(this.TextFont, this.Orig[0], this.Orig[1]);
140 				//Draw Cursor Layer Depending on index
141 			}
142 			break;
143 			default:
144 			break;
145 		}
146 	}else{
147 	alert("Cannot Display the Menu missing required parameters");
148 	}
149 }
