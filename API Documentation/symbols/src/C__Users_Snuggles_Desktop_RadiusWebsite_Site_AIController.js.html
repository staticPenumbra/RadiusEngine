 64 //---------------------------------------------PATHFINDING FUNCTIONS
 65 //Function to move to the last detected X,Y of the Target
 66 /**
 67 * @param {Entity} Target The Entity to move to
 68 * @param {Entity} Entity The moving Entity
 69 */
 70 AIController.prototype.MoveToTarget = function(Target, Entity) {
 71 	if(Target != null && Entity != null){
 72 		var TargetPosition = Target.GetCorners();
 73 		var MoverPosition = Entity.GetCorners();
 74 		var dx = null;
 75 		var dy = null;
 76 		//Calculate midpoint of the Entities
 77 		var TargetMid = new Array(Math.round((TargetPosition[3][0] + TargetPosition[0][0]) / 2) , Math.round((TargetPosition[3][1] + TargetPosition[0][1]) / 2));
 78 		var MoverMid = new Array(Math.round((MoverPosition[3][0] + MoverPosition[0][0]) / 2) , Math.round((MoverPosition[3][1] + MoverPosition[0][1]) / 2));
 79 		
 80 		//Target is in quadrant 1
 81 		if(MoverMid[0] <= TargetMid[0] && MoverMid[1] >= TargetMid[1]){
 82 			if(MoverMid[0] < TargetMid[0] && MoverMid[1] > TargetMid[1]){
 83 				Entity.ApplyVelocity('ne', Entity.GetSpeed());
 84 				Entity.SetMoving(true);
 85 			}else{
 86 				//if x < then increase x by speed
 87 				if(MoverMid[0] < TargetMid[0]){
 88 					Entity.ApplyVelocity('e', Entity.GetSpeed());
 89 					Entity.SetMoving(true);
 90 				//else if y < then increase y by speed
 91 				}else{
 92 					Entity.ApplyVelocity('n', Entity.GetSpeed());
 93 					Entity.SetMoving(true);
 94 				}
 95 			}
 96 		}
 97         //Target is in quadrant 2
 98 		if(MoverMid[0] >= TargetMid[0] && MoverMid[1] >= TargetMid[1]){
 99 			if(MoverMid[0] > TargetMid[0] && MoverMid[1] > TargetMid[1]){
100 				Entity.ApplyVelocity('nw', Entity.GetSpeed());
101 				Entity.SetMoving(true);
102 			}else{
103 				//if x < then increase x by speed
104 				if(MoverMid[0] > TargetMid[0]){
105 					Entity.ApplyVelocity('w', Entity.GetSpeed());
106 					Entity.SetMoving(true);
107 				//else if y < then increase y by speed
108 				}else{
109 					Entity.ApplyVelocity('n', Entity.GetSpeed());
110 					Entity.SetMoving(true);
111 				}
112 			}
113 		}
114 		//Target is in quadrant 3
115 		if(MoverMid[0] >= TargetMid[0] && MoverMid[1] <= TargetMid[1]){
116 			if(MoverMid[0] > TargetMid[0] && MoverMid[1] < TargetMid[1]){
117 				Entity.ApplyVelocity('sw', Entity.GetSpeed());
118 				Entity.SetMoving(true);
119 			}else{
120 				//if x < then increase x by speed
121 				if(MoverMid[0] > TargetMid[0]){
122 					Entity.ApplyVelocity('w', Entity.GetSpeed());
123 					Entity.SetMoving(true);
124 				//else if y < then increase y by speed
125 				}else{
126 					Entity.ApplyVelocity('s', Entity.GetSpeed());
127 					Entity.SetMoving(true);
128 				}
129 			}
130 		}
131 		//Target is in quadrant 4
132 		if(MoverMid[0] <= TargetMid[0] && MoverMid[1] <= TargetMid[1]){
133 			if(MoverMid[0] < TargetMid[0] && MoverMid[1] < TargetMid[1]){
134 				Entity.ApplyVelocity('se', Entity.GetSpeed());
135 				Entity.SetMoving(true);
136 			}else{
137 				//if x < then increase x by speed
138 				if(MoverMid[0] < TargetMid[0]){
139 					Entity.ApplyVelocity('e', Entity.GetSpeed());
140 					Entity.SetMoving(true);
141 				//else if y < then increase y by speed
142 				}else{
143 					Entity.ApplyVelocity('s', Entity.GetSpeed());
144 					Entity.SetMoving(true);
145 				}
146 			}
147 		}
148 	}
149 }
150 //Calculates if the second Entity is within range of the first
151 /**
152 * @param {Entity} Viewer The Entity doing the searching
153 * @param {Entity} Viewee The Entity to compare to
154 * @return {Boolean} Returns true: the target is in range or false: not in range
155 */
156 AIController.prototype.DetectAvailableTarget = function(Viewer, Viewee) {
157 	if(Viewer != null && Viewee != null){
158 		var VieweePoints = Viewee.GetCorners();
159 		var ViewerPoints = Viewer.GetCorners();
160 		var ViewerSightRadius = Viewer.GetSightRadius();
161 		var DistanceBetween = null;
162 		var dx = null;
163 		var dy = null;
164 		//Calculate midpoint of the Entities
165 		var VieweeMid = new Array(Math.round((VieweePoints[3][0] + VieweePoints[0][0]) / 2) , Math.round((VieweePoints[3][1] + VieweePoints[0][1]) / 2));
166 		var ViewerMid = new Array(Math.round((ViewerPoints[3][0] + ViewerPoints[0][0]) / 2) , Math.round((ViewerPoints[3][1] + ViewerPoints[0][1]) / 2));
167 		
168 		//First find what scenario we're dealing with and compare the points in the correct scenario
169 		//D = sqrt((dx^2 + dy^2)
170 		//UR Q1 E1PT2 to E2PT3
171 		if(ViewerMid[0] <= VieweeMid[0] && ViewerMid[1] >= VieweeMid[1]){
172 		//Compare those points, find the positive difference
173 			dx = VieweeMid[0] - ViewerMid[0];
174 			dy = ViewerMid[1] - VieweeMid[1];
175 			DistanceBetween = Math.round(Math.sqrt((dx * dx)+(dy * dy)));
176 		}
177         //UL Q2 E1PT1 to E2PT4
178 		if(ViewerMid[0] >= VieweeMid[0] && ViewerMid[1] >= VieweeMid[1]){
179 		//Compare those points, find the positive difference
180 			dx = ViewerMid[0] - VieweeMid[0];
181 			dy = ViewerMid[1] - VieweeMid[1];
182 			DistanceBetween = Math.round(Math.sqrt((dx * dx)+(dy * dy)));
183 		}
184 		//LL Q3 E1PT3 to E2PT2
185 		if(ViewerMid[0] >= VieweeMid[0] && ViewerMid[1] <= VieweeMid[1]){
186 		//Compare those points, find the positive difference
187 			dx = ViewerMid[0] - VieweeMid[0];
188 			dy = VieweeMid[1] - ViewerMid[1];
189 			DistanceBetween = Math.round(Math.sqrt((dx * dx)+(dy * dy)));
190 		}
191 		//LR Q4 E1PT4 to E2PT1
192 		if(ViewerMid[0] <= VieweeMid[0] && ViewerMid[1] <= VieweeMid[1]){
193 		//Compare those points, find the positive difference
194 			dx = VieweeMid[0] - ViewerMid[0];
195 			dy = VieweeMid[1] - ViewerMid[1];
196 			DistanceBetween = Math.round(Math.sqrt((dx * dx)+(dy * dy)));
197 		}
198 		//Compare to view radius
199 		if(DistanceBetween <= ViewerSightRadius){
200 			return(true);
201 		}else{
202 			return(false);
203 		}	
204 	}
205 }

207 //Targets the player nearest the entity if they are in view
208 /**
209 * @return {Entity} Returns the current entity being processed or null if none in range
210 */
211 AIController.prototype.TargetNearestPlayer = function(Entity) {
212 	if(this.Entities != null){
213 		//-----SECTION 1 VARIABLES
214         var TargetSelection = null;
215 		var PlayersDetected = new Array();
216 		//Gather a reference to the player entities
217         for(var i = 0; i < this.Entities.length; i++){
218             if(this.Entities[i].GetType() == "player1" || this.Entities[i].GetType() == "player2"){
219 				PlayersDetected.push(this.Entities[i]);
220             } 
221         }	
222 		//update the player list
223 		this.PlayerEntities = PlayersDetected;
224 		//-----SECTION 2 VARIABLES
225 		var NumPlayerEntities = this.PlayerEntities.length;
226 			
227         //Consider who is available and closest if there are potential targets
228         if(NumPlayerEntities != 0){
229             for(var i = 0; i < NumPlayerEntities; i++){
230                 //Decide if they are an available target
231 				var GoodTarget = this.DetectAvailableTarget(Entity, this.PlayerEntities[i]);
232 				//Target the entity if they are in range	
233 				if(GoodTarget == true){
234 					Entity.SetCurrentTarget(this.PlayerEntities[i]);
235 					return(this.PlayerEntities[i]);
236 				}
237 			}
238 		return(null);	
239 		}
240 	}
241 }

246 //Runtime Routine for a zombie entity
247 AIController.prototype.SimpleZombieRoutine = function() {
248 	//Target the nearest Player if in view
249 	if(this.PlayerEntities != null){
250 		for(var i = 0; i < this.Entities.length; i++){
251 			//If the entity is a zombie then target nearest player
252 			if(this.Entities[i].GetType() == "zombie"){
253 				var newTarget = this.TargetNearestPlayer(this.Entities[i]);
254 				this.MoveToTarget(newTarget , this.Entities[i]);
255 			}
256 		}
257 	}
258 }

260 //AI routine for the Pistol Bullet
261 AIController.prototype.PistolBulletRoutine = function() {
262     if(this.Entities != null){
263     var CurrentEntityPos = this.CurrentEntity.GetPosition();
264     var CurrentVelocity = this.CurrentEntity.GetVelocity();
265     
266     //direction magnitude
267         switch(CurrentVelocity[0]){
268         case "n":
269         this.CurrentEntity.ApplyVelocity('n', this.CurrentEntity.GetSpeed());
270 		this.CurrentEntity.SetMoving(true);
271         break;
272         case "s":
273         this.CurrentEntity.ApplyVelocity('s', this.CurrentEntity.GetSpeed());
274 		this.CurrentEntity.SetMoving(true);
275         break;
276         case "w":
277         this.CurrentEntity.ApplyVelocity('w', this.CurrentEntity.GetSpeed());
278 		this.CurrentEntity.SetMoving(true);
279         break;
280         case "e":
281         this.CurrentEntity.ApplyVelocity('e', this.CurrentEntity.GetSpeed());
282 		this.CurrentEntity.SetMoving(true);
283         break;
284         case "c":
285 		this.CurrentEntity.SetMoving(false);
286         break;
287         default:
288         break;
289         }
290     }

293 //Runs the AI routine for the specified Entity
294 /**
295 * @param {Entity} Entity The Entity to run an AI routine on
296 */
297 AIController.prototype.RunRoutine = function(Entity) {
298     if(Entity != null && Entity != 0){
299         var tempType = Entity.GetType();
300         switch(tempType){
301             case "player1":
302             break;
303             case "player2":
304             break;
305             case "zombie":
306             this.SimpleZombieRoutine();
307             break;
308             case "TVZ_pistolBullet":
309             this.PistolBulletRoutine();
310             break;
311             case "TVZ_shotgunBullet":
312             break;
313             case "TVZ_rifleBullet":
314             break;
315             default:
316 				alert("AI Error: Unlisted entity type");
317             break;
318         }   
319     }
