 17 /**
 18  * @class
 19  * Class modeling a event Trigger
 20  *
 21  * @description
 22  * Event Triggers are walkover and scripted things that happen
 23  **/
 24  /**
 25  * @constructor
 26  */
 27 var Trigger = function(Type, FireOnce, Position, Dimensions) {
 28 	this.Dimensions = Dimensions; //Dimension dimension array for the trigger (width, height)
 29 	this.Position = Position; //Position array for the trigger (x, y)
 30 	this.FireOnce = FireOnce; //Fireonce boolean flag true=once false=infinite
 31 	this.Type = Type; //Text string for the trigger typep
 32 	this.TrippedBy = null;
 33 	this.UL = Position;
 34 	this.UR = new Array(this.UL[0] + this.Dimensions[0], this.UL[1]);
 35 	this.LL = new Array(this.UL[0], this.UL[1] + this.Dimensions[1]);
 36 	this.LR = new Array(this.UL[0] + this.Dimensions[0], this.UL[1] + this.Dimensions[1]);
 37 }
 38 //------------------------------------------------------GET ACCESSORS-----------------------------------------------
 39 /**
 40 * Returns the last entity to trip the trigger
 41 * @returns {Entity} The entity that last tripped the trigger
 42 */
 43 Trigger.prototype.GetTrippedBy = function(){
 44 	if(this.TrippedBy != null){
 45 		return(this.TrippedBy);
 46 	}
 47 }
 48 /**
 49 * Returns an array of the triggers 4 corner points
 50 * @returns {Array[]} An array of 4 points, one for each corner of the Trigger
 51 */
 52 Trigger.prototype.GetCorners = function(){
 53 	return(new Array(this.UL, this.UR, this.LL, this.LR));
 54 }
 55 /**
 56 * Returns the current type of the trigger
 57 * @returns {String} The current type of the trigger
 58 */
 59 Trigger.prototype.GetType = function(){
 60 	if(this.Type != null){
 61 		return(this.Type);
 62 	}
 63 }
 64 /**
 65 * Returns the whether the current trigger is fire once or not
 66 * @returns {Boolean} The current trigger state (true) fire once or (false) trigger resets
 67 */
 68 Trigger.prototype.GetFireOnce = function(){
 69 	if(this.FireOnce != null){
 70 		return(this.FireOnce);
 71 	}
 72 }
 73 /**
 74 * Returns the upper left hand position of the trigger
 75 * @returns {Array[]} The point Array X/Y for the upper left hand position
 76 */
 77 Trigger.prototype.GetPosition = function(){
 78 	if(this.Position != null){
 79 		return(this.Position);
 80 	}
 81 }
 82 /**
 83 * Gets the current trigger dimensions
 84 * @returns {Array[]} The point Array width/height indicating the current trigger dimensions
 85 */
 86 Trigger.prototype.GetDimensions = function(){
 87   if(this.Dimensions != null){
 88 	return(this.Dimensions);
 89   }
 90 }
 91 //---------------------------------------------------------------SET ACCESSORS------------------------------------------
 92 /**
 93 * Sets the current trigger to fire once(true) or resets(false)
 94 * @param {Boolean} Boolean true or false , fire once or reset to fire again
 95 */
 96 Trigger.prototype.SetFireOnce = function(Boolean){
 97 	if(Boolean != null){
 98 		this.FireOnce = Boolean;
 99 	}
100 }
101 /**
102 * Sets the upper left hand point position for the current trigger
103 * @param {Array[]} Position Array with position information X/Y
104 */
105 Trigger.prototype.SetPosition = function(Position){
106 	if(Position != null){
107 		this.Position = Position;
108 	}
109 }
110 /**
111 * Sets trigger dimensions
112 * @param {Array[]} Dimensions Array with width/height information
113 */
114 Trigger.prototype.SetDimensions = function(Dimensions){
115 	if(Dimensions != null){
116 		this.Dimension = Dimensions;
117 	}
118 }
119 
120 //-------------------------------------------------------------UTILITY FUNCTIONS-------------------------------------------
121 /**
122 * Sets the triggering entity and returns a reference to the current trigger
123 * @param {Entity} Entity reference to the tripping Entity
124 * @returns {Trigger} The current Trigger
125 */
126 Trigger.prototype.Fire = function(Entity){
127 	if(Entity != null){
128 		this.TrippedBy = Entity;
129 		return(this);
130 	}
131 }
