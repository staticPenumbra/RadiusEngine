 16 /**
 17  * @class
 18  * Class modeling the database interface to application resources
 19  *
 20  * @description
 21  * All resources are aquired and manipulated through this manager in order to provide a clear simple way to access them
 22  **/
 23  /**
 24  * @constructor
 25  */
 26 var ResourceManager = function(Root) {
 27 	this.Pages = new Array(new Stage("Index", Root));
 28 	this.Pages[0].Load();
 29 	this.ResourceRoot = Root;
 30 	/*-------------------DATABASE CACHE---------------------------*/
 31 	this.L2DOMCache = new Array();
 32 	this.L2ImageCache= new Array();
 33 	this.L2AudioCache = new Array();
 34 	this.CacheInit(1);
 35 }
 36 //-----------------------------------------------------Get Accessors----------------------------------------------
 37 /**
 38 * Initialize or Reset the L2 Cache to Page defaults
 39 * @param {Integer} PageIndex The page to load from
 40 */
 41 ResourceManager.prototype.CacheInit = function(PageIndex){
 42 	if(PageIndex != null && PageIndex > 0){
 43 	//Init DOM
 44 		this.L2DOMCache = this.GetDOMDefault(PageIndex);
 45 	//Init Images
 46 		this.L2ImageCache = this.LoadBackgrounds(PageIndex);
 47 	//Init Audio
 48 		this.L2AudioCache = this.LoadPageAudio(PageIndex);
 49 	}
 50 }
 51 /**
 52 * Add a DOM element to L2 Application cache
 53 * @param {Array} Value The Array of DOM data for that element
 54 */
 55 ResourceManager.prototype.AddDOMElement = function(Value){
 56 	if(Value != null){
 57 		this.L2DOMCache.push(Value);
 58 	}
 59 }
 60 /**
 61 * Return the Current DOM
 62 * @param {Integer} PageIndex The page to load from
 63 * @return {Array[]} Returns an array of DOM elements
 64 */
 65 ResourceManager.prototype.GetDOM = function(PageIndex){
 66 	if(PageIndex != null && PageIndex > 0){
 67 		return(this.L2DOMCache);
 68 	}
 69 }
 70 /**
 71 * Return the Current Images
 72 * @param {Integer} PageIndex The page to load from
 73 * @return {Image[]} Returns an array of Image elements
 74 */
 75 ResourceManager.prototype.GetImages = function(PageIndex){
 76 	if(PageIndex != null && PageIndex > 0){
 77 		return(this.L2ImageCache);
 78 	}
 79 }
 80 /**
 81 * Return the Current Audio
 82 * @param {Integer} PageIndex The page to load from
 83 * @return {Audio[]} Returns an array of Audio elements
 84 */
 85 ResourceManager.prototype.GetAudio = function(PageIndex){
 86 	if(PageIndex != null && PageIndex > 0){
 87 		return(this.L2AudioCache);
 88 	}
 89 }
 90 /**
 91 * Return the starting position of the specified DOM Entity of the specified type
 92 * @param {Integer} PageIndex The page to load from
 93 * @return {Integer[]} Returns an array of DOM elements
 94 */
 95 ResourceManager.prototype.GetDOMDefault = function(PageIndex){
 96 	if(PageIndex != null && PageIndex > 0){
 97 		return(this.Pages[PageIndex-1].DOMData);
 98 	}
 99 }
100 /**
101 * Return the starting position of the specified Entity of the specified type
102 * @param {Integer} PageIndex The page to load from
103 * @return {Integer[]} Returns an array specifying the x and y dimensions of the specified page
104 */
105 ResourceManager.prototype.GetPageDimensions = function(PageIndex){
106 	if(PageIndex != null && PageIndex > 0){
107 		return(this.Pages[PageIndex-1].GetDimensions());
108 	}
109 }
110 /**
111 * Load all page triggers from the database and return an array of them
112 * @param {Integer} PageIndex The page to load from
113 * @return {Array[]} Returns an array of Triggers
114 */
115 ResourceManager.prototype.LoadTriggers = function(PageIndex){
116 	if(PageIndex != null && PageIndex > 0){
117 		var Trigs = new Array();
118 		for(var i = 0; i < this.Pages[PageIndex-1].Triggers.length; i++){
119 			//Type, FireOnce, Position, Dimensions
120 			var add = this.GetEntityData(PageIndex, this.Pages[PageIndex-1].Triggers[i]);
121 			//"dialogue", true, 200, 200, 50, 50
122 			Trigs.push(new Trigger(add[0], add[1], new Array(add[2], add[3]), new Array(add[4], add[5])));
123 		}
124 		return(Trigs);
125 	}
126 }
127 /**
128 * Return the starting position of the specified Entity of the specified type
129 * @param {Integer} PageIndex The page to load from
130 * @param {String} EntityKey The database name of the entity
131 * @return {Integer[]} Returns an array specifying the x and y coordinate of the entity starting position
132 */
133 ResourceManager.prototype.GetEntityData = function(PageIndex, EntityKey){
134 	if(PageIndex != null && PageIndex > 0 && EntityKey != null){
135 		return(this.Pages[PageIndex-1].CacheRetrieve("Entity", EntityKey));
136 	}
137 }
138 /**
139 * Returns an audio object for the stage audio
140 * @param {Integer} PageIndex Page number to test
141 * @return {Boolean} Returns a flag indicating "true" this is the index page or "false" it isn't
142 */
143 ResourceManager.prototype.IsTitleScreen = function(PageIndex){
144 	if(PageIndex != null && PageIndex > 0){
145 		return(this.Pages[PageIndex-1].TitleFlag);
146 	}
147 }
148 //----------------------------------------------------Set Accessors------------------------------------------------
149 //Warning: DATABASE DEFAULT SHOULD BE READ ONLY!!
150 //----------------------------------------------------UTILITY FUNCTIONS-------------------------------------------
151 /**
152 * Returns an audio object for the stage audio
153 * @param {Integer} PageIndex Page number to load from 
154 * @return {Array[]} Returns a loaded array of the pages Audio
155 */
156 ResourceManager.prototype.LoadPageAudio = function(PageIndex){
157 	if(PageIndex != null && PageIndex > 0){
158 		var toReturn = new Array();
159 		//create an Audio object for each sound in the Page
160 		for(var i =0; i < this.Pages[PageIndex-1].AudioKeys.length; i++){
161 			var temp = this.Pages[PageIndex-1].CacheRetrieve("Audio", this.Pages[PageIndex-1].AudioKeys[i]);
162 			toReturn.push(new Audio(temp, true, 5, 10.0, ".ogg"));
163 		}
164 		return(toReturn);
165 	}
166 }
167 /**
168 * Caches and returns all backgrounds for the specified stage
169 * @param {Integer} PageNumber Page number to load from 
170 * @return {Image[]} Returns all background objects for the current stage
171 */
172 ResourceManager.prototype.LoadBackgrounds = function(PageNumber){
173 	if(PageNumber != null){
174 		var thePage = this.Pages[PageNumber - 1];
175 		var RetrievedBackgrounds = new Array();
176 		
177 		for(var i = 0; i < thePage.Backgrounds.length; i++){
178 			RetrievedBackgrounds.push(thePage.CacheRetrieve("Image", thePage.Backgrounds[i]));
179 		}
180 		return(this.ImageLoader(RetrievedBackgrounds));
181 	}
182 }
183 /**
184 * Takes an array of filenames and returns the corresponding array of Image objects
185 * @param {String[]} FilenameArray The array of filename strings to load
186 * @return {Image[]} Returns a reference to the loaded Images in array format
187 */
188 ResourceManager.prototype.ImageLoader = function(FilenameArray){
189     if(FilenameArray != null){
190         var ReturnArray = new Array();
191         for(var i = 0; i < FilenameArray.length; i++){
192                 ReturnArray.push(new Image());
193                 ReturnArray[i].src = FilenameArray[i];
194         }
195         return(ReturnArray);
196     }
197 }
