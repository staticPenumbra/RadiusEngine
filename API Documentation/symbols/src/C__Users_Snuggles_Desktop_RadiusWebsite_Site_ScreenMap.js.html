 17 /**
 18  * @class
 19  * Class modeling a current display frame and objects rendered to screen
 20  *
 21  * @description
 22  * An object describing all of the information about a display frame
 23  **/
 24 /**
 25 * @constructor
 26 */
 27 var ScreenMap = function(ResolutionX, ResolutionY, FrontCanvasContext, RearCanvasContext) {
 28     //Default Constructor
 29 	this.bctx = RearCanvasContext; //The canvas context for the background screen buffer
 30 	this.ctx = FrontCanvasContext; //The canvas context for the foreground screen buffer
 31 	this.XResolution = ResolutionX; //Screen resolution X value in integer format
 32 	this.YResolution = ResolutionY; //Screen resolution Y value in integer format
 33 	this.Entities = new Array();	//The Array of current Application entities
 34 	this.BackgroundImages = new Array(); //The Current array of background Images
 35 	this.ZoomLevel = null;		//Current screen magnification level 
 36 	this.MenuSystem = new Array(); //Array of Menu layouts for the stage
 37 	this.DOM = new Array();
 38 }
 39 
 40 //-----------------------------------------------------Get Methods-----------------------------------
 41 //Function to get the state of the menu option pointer
 42 /**
 43 *@return {Integer} Returns an integer value for the state of the menu pointer
 44 */	
 45 ScreenMap.prototype.GetMenuPointerPosition = function(){
 46 	return(this.MenuSystem[0].GetCursorPosition());
 47 }
 48 //Function to return the first Menu in the menu system
 49 /**
 50 *@return {ApplicationMenu} Returns the first menu in the list of current stage menus
 51 */	
 52 ScreenMap.prototype.GetMenu = function(){
 53 	if(this.MenuSystem[0] != null){
 54 		return(this.MenuSystem[0]);
 55 	}
 56 }
 57 //-----------------------------------------------------Set Methods-----------------------------------
 58 //Function to set the state of the menu option pointer
 59 /**
 60 * @param {Integer} Option The menu option to switch the cursor to
 61 */
 62 ScreenMap.prototype.SetMenuPointerPosition = function(Option){
 63 	if(Option != null && Option >= 0 && Option < this.MenuSystem[0].MenuOptions.length){
 64 		this.MenuSystem[0].ChangeCursorPosition(Option);
 65 	}
 66 }
 67 //Sets the current Menu system
 68 /**
 69 * @param {ApplicationMenu} Menu Adds the specified menu to the menu system for the stage, if passed null stops displaying the menu
 70 */
 71 ScreenMap.prototype.SetMenuSystem = function(Menu){
 72 	if(Menu != null){
 73 		this.MenuSystem = new Array();
 74 		this.MenuSystem.push(Menu);
 75 	}else{
 76 	//Escape pressed again stop displaying menu
 77 		this.MenuSystem = new Array();
 78 	}
 79 }

 81 /**
 82 * @param {Image} BG An instance of the background image to set
 83 */
 84 ScreenMap.prototype.SetBackgrounds = function(BG){
 85 	if(BG != null){
 86 		this.BackgroundImages = BG;
 87 	}
 88 }
 89 //Function to set the screenmap resolution
 90 /**
 91 * @param {Integer} ResolutionX The value for the x component of the display resolution
 92 * @param {Integer} ResolutionY The value for the y component of the display resolution
 93 */
 94 ScreenMap.prototype.SetResolution = function(ResolutionX, ResolutionY){
 95     if(ResolutionX != null && ResolutionY != null){
 96     this.XResolution = ResolutionX;
 97     this.YResolution = ResolutionY;
 98     }
 99 }
100 //----------------------------------------------------Render Functions----------------------------------

102 //Clears the current screen not needed with blitting
103 ScreenMap.prototype.Clear = function(){
104 	this.bctx.clearRect(0, 0, this.XResolution, this.YResolution);
105 }
106 //Function to flip the displayed screens by swapping video buffer
107 ScreenMap.prototype.Blit = function(){
108 	if(this.bctx != null && this.ctx != null){
109 		var offscreen_data = this.bctx.getImageData(0, 0, this.XResolution, this.YResolution);
110 		this.ctx.putImageData(offscreen_data, 0, 0);
111 	}
112 }
113 //Function to render the map to the background canvas and blit
114 ScreenMap.prototype.RenderToCanvas = function(){
115        this.bctx.drawImage(this.BackgroundImages[0], 0, 0);
116 	   if(this.BackgroundImages.length > 1){
117        this.bctx.drawImage(this.BackgroundImages[1], 0, 0);
118 	   }
119        //Render Entities
120        //Make sure there are entities
121 	   if(this.DOM != null){
122 		 for(var i = 0; i <= this.DOM.length - 1; i++){
123            //ALL LINKS FOR NOW
124 		    this.bctx.fillStyle = "white";
125 			this.bctx.font = this.DOM[i][5];
126 			this.bctx.fillText(this.DOM[i][4], this.DOM[i][0], this.DOM[i][1]);
127         } 
128 	   }
129 	   
130        if(this.Entities != null){
131         for(var i = 0; i <= this.Entities.length - 1; i++){
132             if(this.Entities[i].SpriteSheet != null){
133                 //We have an Image to display so show it
134                 //Render Sprites
135                 this.DrawSprite(this.bctx, this.Entities[i]);
136             }
137         }
138        }
139 	   if(this.BackgroundImages.length > 2){
140        this.bctx.drawImage(this.BackgroundImages[2], 0, 0);
141 	   }
142     //}
143 	//Account for a closed menu and a freshly constructed system
144 	if(this.MenuSystem != null && this.MenuSystem.length != 0){
145 		for(var i = 0; i < this.MenuSystem.length; i++){
146 			this.MenuSystem[i].Display(this.bctx);
147 		}
148 	}
149 	//Blit the image to screen
150 	this.Blit();
151 }
152 //Render Event; Updates Entities and renders to canvas
153 /**
154 * @param {Entity[]} EntityList Updated Entity List to process
155 * @param {Array[]} DOMList Updated DOM List to process
156 */
157 ScreenMap.prototype.RenderCycle = function(EntityList, DOMList) {
158     this.UpdateDOM(DOMList);
159 	this.UpdateEntities(EntityList);
160     this.RenderToCanvas();
161 }
162 //Function to draw a specified Entity on the background canvas for blitting
163 /**
164 * @param {CanvasContext} bctx Canvas context reference to the background canvas
165 * @param {Entity} Entity A reference to the entity to render
166 */
167 ScreenMap.prototype.DrawSprite = function(bctx, Entity){
168      if(bctx != null && Entity != null){
169 		//img,x,y,width,height
170 		var position = Entity.GetPosition();
171 		var xcord = position[0];
172 		var ycord = position[1];
173 		var Dimensions = Entity.GetImageDimensions();
174 		var width = Dimensions[0];
175 		var height = Dimensions[1];
176 		var Vec = Entity.GetVelocity();
177 		var Sprite = Entity.GetSpriteSheet();
178 		var CurrentFrame = Entity.GetFrame();
179 		var EntType = Entity.GetType();
180 		//-----------------------------------------Rendering Rules--------------------------------
181 		switch(EntType){
182 			case "TVZ_pistolBullet":
183 				bctx.drawImage(Sprite, xcord, ycord, width, height);
184 			break;
185 			case "TVZ_rifleBullet":
186 				bctx.drawImage(Sprite, xcord, ycord, width, height);
187 			break;
188 			case "TVZ_shotgunBullet":
189 				bctx.drawImage(Sprite, xcord, ycord, width, height);
190 			break;
191 			case "Wall":
192 				bctx.drawImage(Sprite, xcord, ycord, width, height);
193 			break;
194 			case "Door":
195 				bctx.drawImage(Sprite, xcord, ycord, width, height);
196 			break;
197 			default:
198 				if(EntType == "player1" || EntType == "player2" || EntType == "zombie"){
199 					//Display a different image based upon the direction of travel
200 					switch(Vec[0]){
201 						case 'n':
202 							//xrow = 5, 4 cells
203 							this.Animate('n', Entity, bctx);
204 						break;
205 						case 'ne':
206 							//xrow = 0, 4 cells
207 							this.Animate('ne', Entity, bctx);
208 						break;
209 						case 'e':
210 							//xrow = 6, 6 cells
211 							this.Animate('e', Entity, bctx);
212 						break;
213 						case 'se':
214 							//xrow = 1, 4 cells
215 							this.Animate('se', Entity, bctx);
216 						break;
217 						case 's':
218 							//xrow = 4, 4 cells
219 							this.Animate('s', Entity, bctx);
220 						break;
221 						case 'sw':
222 							//xrow = 3, 4 cells
223 							this.Animate('sw', Entity, bctx);
224 						break;
225 						case 'w':
226 							//xrow = 7, 6 cells
227 							this.Animate('w', Entity, bctx);
228 						break;
229 						case 'nw':
230 							//xrow = 2, 4 cells
231 							this.Animate('nw', Entity, bctx);
232 						break;
233 						case 'c':
234 							Entity.SetFrame(0);
235 							bctx.drawImage(Sprite,0, 128, width, height, xcord, ycord, width, height);
236 						break;
237 						default:
238 							alert("Screenmap line 168 cannot draw image");
239 						break;
240 					}
241 				}
242 			break;
243 		}
244     }
245 }

269 //Function to animate an entity and draw the correct animation frame 
270 /**
271 * @param {String} Direction String indicating the walking animation direction
272 * @param {Entity} Entity A reference to the entity to render
273 * @param {Entity} bctx A reference to the background canvas context
274 */
275 ScreenMap.prototype.Animate = function(Direction, Entity, bctx){
276 	if(Direction != null && Entity != null && bctx != null){
277 		var Cells = Entity.GetCells(Direction);
278 		var position = Entity.GetPosition();
279 		var xcord = position[0];
280 		var ycord = position[1];
281 		var Sprite = Entity.GetSpriteSheet();
282 		var CurrentFrame = Entity.GetFrame();
283 		var ID = Entity.GetImageDimensions();
284 		var AnimationDelay = 80;
285 		var CellOrder = Entity.GetCellOrder();
286 		switch(Direction){
287 				case 'n':
288 					var Offset = 0;
289 				break;
290 				case 'ne':
291 					var Offset = 1;
292 				break;
293 				case 'e':
294 					var Offset = 2;
295 				break;
296 				case 'se':
297 					var Offset = 3;
298 				break;
299 				case 's':
300 					var Offset = 4;
301 				break;
302 				case 'sw':
303 					var Offset = 5;
304 				break;
305 				case 'w':
306 					var Offset = 6;
307 				break;
308 				case 'nw':
309 					var Offset = 7;
310 				break;
311 		}
312 		if(Entity.IsAnimating() == true){
313 			if(CurrentFrame >= Cells){
314 				Entity.SetFrame(0);
315 			}
316 			if(CurrentFrame < Cells){
317 				bctx.drawImage(Sprite,CurrentFrame * ID[0], ID[1] * CellOrder[Offset], ID[0], ID[1], xcord, ycord, ID[0], ID[1]);
318 				//Update the frame only if a certain amount of time has passed
319 				var previous = Entity.GetTimestamp();
320 				var next = Math.round(new Date().getTime());
321 				if(previous + AnimationDelay < next){
322 					CurrentFrame++;
323 					if(CurrentFrame >= Cells){
324 						Entity.SetFrame(0);
325 						Entity.UpdateModified();
326 					}else{
327 						Entity.SetFrame(CurrentFrame);
328 						Entity.UpdateModified();
329 					}
330 				}
331 			}
332 		}else{
333 			Entity.SetFrame(0);
334 			bctx.drawImage(Sprite,CurrentFrame * ID[0], ID[1] * CellOrder[Offset], ID[0], ID[1], xcord, ycord, ID[0], ID[1]);
335 		}
336 	}
