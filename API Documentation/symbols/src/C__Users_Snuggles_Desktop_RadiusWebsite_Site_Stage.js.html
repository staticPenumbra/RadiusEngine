 26 var Stage = function(Name, fileRoot) {

 28 	this.ResetDatabase();

 30     this.Name = Name; //Page or Area reference name

 97 //Searches Cache for Item and returns if present otherwise runs retrieve and load then returns item
 98 /**
 99 * @param {String} RootContainer Name of the root container to search in
100 * @param {String} Key Key value String to search for and return
101 * @return {Array[]} Returns a reference to the container searched for
102 */
103 Stage.prototype.CacheRetrieve = function(RootContainer, Key){
104 	if(RootContainer != null && Key != null){
105 		var CachePull = this.CacheKeys.indexOf(Key);
106 		if(CachePull == -1){
107 			//Cache miss
108 			//Try the pull from Database 
109 			var DBPull = this.Retrieve(RootContainer, Key);
110 			if(DBPull == -1){
111 				//Item does not exist in cache or Database
112 				alert("Element: " + Key + " not found in database during cache retrieval");
113 			}else{
114 				//Add the element to the cache and then return
115 				this.CacheKeys.push(Key);
116 				this.CacheData.push(DBPull);
117 				return(DBPull);
118 			}
119 		}else{
120 			//found it, return database element
121 			return(this.CacheData[CachePull]);
122 		}
123 	}
124 }
125 //Locates Declaration and loads then returns loaded
126 /**
127 * @param {String} RootContainer Name of the root container to search in
128 * @param {String} Key Key value String to search for and return
129 * @return {Array[]} Returns a reference to the container searched for
130 */
131 Stage.prototype.Retrieve = function(RootContainer, Key){
132 	if(RootContainer != null && Key != null){
133 		switch(RootContainer){
134 			case "Audio":
135 				var KeyRoot = this.AudioKeys;
136 				var DataRoot = this.AudioData;
137 			break;
138 			case "Image":
139 				var KeyRoot = this.ImageKeys;
140 				var DataRoot = this.ImageData;
141 			break;
142 			case "DOM":
143 				var KeyRoot = this.DOMKeys;
144 				var DataRoot = this.DOMData;
145 			break;
146 			case "Entity":
147 				var KeyRoot = this.EntityKeys;
148 				var DataRoot = this.EntityData;
149 			break;
150 			default:
151 				alert("Error 101(47z: Database retrieval -- Unknown root element");
152 			break;
153 		}
154 		//Search the Key string
155 		var index = KeyRoot.indexOf(Key);
156 		if(index == -1){
157 			//DB miss
158 			alert("Element: " + Key + " not found in database");
159 			return(-1);
160 		}else{
161 			//found it, return database element
162 			return(DataRoot[index]);
163 		}
164 	//Error in processing return null
165 	return(null);
166 	}
167 }
