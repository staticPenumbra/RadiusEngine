 30

 33     this.CreatureType = TVZ_CreatureType; //String name of the current creature type

 35 	this.PlayerInput = null;  //The player Input object (if defined then PC not defined is NPC)

 40 	this.HitPoints = 1; //The total number of "hits" the entity can withstand before destruction **determined by creature type
 41 	this.Speed = 0; //The traveling speed of the Entity **determined by creature Type
 42 	this.TravelDirection = null; //The current traveling direction of the entity
 43 	this.SightRadius = 5; //The sight radius of the current Entity **determined by creature type
 44 	this.XPos = PosX;  //The current X position of the Entity
 45 	this.YPos = PosY;  //The current Y position of the Entity
 46 	this.ImageWidth = null;   //The Image width for the current spritesheet cell **determined by creature type

 48 	this.DestroySounds = null; //Creature Audio object for death sounds **determined by creature type

 50 	this.AttackSounds = null; //Creature Audio object for attacking sounds **determined by creature type
 51 	this.Faction = TVZ_Faction; //The current faction of the Entity (Determined who can damage and who can be damaged by the current)
 52 	this.Weapon = null;  //The current weapon of the Entity **determined by creature type
 53 	this.Inventory = null; //A String list of current Entities contained in the current Entity inventory **determined by creature type
 54 	this.Acceleration = 0; //Acceleration constant applied to the current entity
 55 	this.VelocityVector = new Array('c', 0); //Velocity vector of the entity indicating both a direction and a magnitude

 57 	this.zoomLevel = 1;    //Zoom level of the Entity for resizing
 58 	this.shadow = 0;    //Shadow on(1) or off(0)
 59 	this.AIFocus = null;   //The current AI target of the object **Set by AI controller

357 //Removes the specified entity from the current entities inventory
358 /**
359 * @param {Entity} Entity Entity to remove from the current Entities inventory
360 */
361 Entity.prototype.RemoveFromInventory = function(Entity) {
362     //Loop through the Inventory, make sure it's created
363     if(this.Inventory != null){
364         //find each instance with the specified tag and delete it
365         for(var j = 0; j < this.Inventory.length; j++){
366             if(this.Inventory[j] == Entity){
367             this.Inventory.splice(j, 1);
368             }
369         }
370     }	
371 }

384 //Sets the type of weapon for the current entity
385 /**
386 * @param {Entity} GunType Inventory gun to change to
387 */
388 Entity.prototype.ChangeWeapon = function(GunType) {
389 	try{
390 	   //Make sure the inventory has been created
391 	   if(this.Inventory != null){
392             for(i = 0; i < this.Inventory.length; i++){
393             //For each Item in the inventory check for the specified gun
394                 if(this.Inventory[i] == GunType){
395                     //Found it so set to be the current weapon
396                     this.Weapon = GunType;
397                 }
398             }
399             if(this.Weapon != GunType){
400             //oops the weapon isn't in the player inventory generate error
401             alert(this.ReferenceName + "Error: 101(47z " + "Cannot change weapon, does not exist in Creature inventory");
402             }
403 	   }
404 	   else{
405 	   alert("Inventory is NULL");
406 	   }
407 	}
408 	catch(err)
409 	{
410 	   alert(this.ReferenceName + "Error: 101(47z: Could not change the current weapon"); 
411 	}
412 }

659 //Function to set up a unique GUID on Entity creation
660 Entity.prototype.Setup = function(){
661     try{
662         switch(this.CreatureType){
663 		case "Door":
664 			this.DoorConfig();
665 			break;
666 			
667 		case "Wall":
668 			this.WallConfig();
669 			break;
670 			
671         case "zombie":
672             this.ZombieConfiguration();
673             break;
674     
675         case "player1":
676             this.PlayerConfiguration();
677             break;
678             
679         case "player2":
680             this.PlayerConfiguration();
681             break;
682             
683         case "TVZ_pistolBullet":
684             this.PistolBConfig();
685             break;
686             
687         case "TVZ_shotgunBullet":
688             this.ShotgunBConfig();
