 16 /**
 17  * @class
 18  * Class modeling an Event
 19  *
 20  * @description
 21  * All in-application events are modeled as a generic event type and then passed to the event manager
 22  **/
 23  /**
 24  * Defines an in application event
 25  * @constructor
 26  */
 27 var EngineEvent = function(Type, Args) {
 28 	this.ApplyFrom = null;
 29 	this.ApplyTo = null;
 30 	this.EventType = Type; //This is the type of event to switch handler based upon
 31 	this.InstanceAtX = null;
 32 	this.InstanceAtY = null;
 33 	this.ApplicationStop = null; //For errors, symbolizes a critical system error(To be avoided)
 34 	this.args = Args; //Arguments to the event handler
 35 }
 36 //---------------------------------------------------------------------------GET ACCESSORS---------------------------------------------------
 37 /**
 38 * Retrieves the event arguments
 39 * @return {Array[]} Stored string arguments for the event
 40 */
 41 EngineEvent.prototype.GetArg = function() {
 42     return(this.args);
 43 }
 44 /**
 45 * Accessor method to get the type of event
 46 * @return {String} Returns Type String of the object
 47 */
 48 EngineEvent.prototype.GetType = function() {
 49     return(this.EventType);
 50 }
 51 /**
 52 * keyUp Event
 53 * @return {KeyCode} Returns keycode passed into it for clojure
 54 */
 55 EngineEvent.prototype.keyUp = function() {
 56      return(this.args);
 57 }
 58 /**
 59 * keyDown Event
 60 * @return {KeyCode} Returns keycode passed into it for clojure
 61 */
 62 EngineEvent.prototype.keyDown = function() {
 63      return(this.args);
 64 }
 65 //--------------------------------------------------------------------------SET ACCESSORS----------------------------------------------------
 66 //--------------------------------------------------------------------------UTILITY FUNCTIONS------------------------------------------------
 67 /**
 68 * Specifies how different events should behave when they have their execute methods called
 69 */
 70 EngineEvent.prototype.Execute = function() {
 71     switch(this.EventType){
 72         //switch handler based upon the type of event
 73         //Specifies a render event
 74         case "render":
 75 			return("render");
 76         break;
 77 		case "trigger":	
 78 		break;
 79         case "pauseAudio":
 80 			alert("Pause Audio");
 81         break;
 82         case "playAudio":
 83 			alert("Play Audio");
 84         break;
 85         case "displaymenu":
 86 			alert("Display Menu");
 87         break;
 88         case "unpauseinterrupt":
 89 			alert("unpauseinterrupt");
 90         break;
 91         case "pauseinterrupt":
 92 			alert("pauseinterrupt");
 93         break;
 94         case "error":
 95 			alert("Application error " + this.args);
 96         break;
 97         case "create":
 98 			return(this.CreateEntity());
 99         break;
100 		case "destroy":
101 			return("killme");
102         break;
103         case "shoot":
104 			alert(this.ApplyFrom + " Shooting");
105         break;
106         case "collision":
107 			alert("Collision" + this.ApplyFrom + " " + this.ApplyTo);
108         break;
109         case "keyUp":
110 			return(this.keyUp());
111         break;
112         case "keyDown":
113 			return(this.keyDown());
114         break;
115         default:
116 			alert("Not a valid event type in Event object");
117         break;
118     }	
119 }
120 /**
121 * Create and return an Entity object
122 * @param {String} TYPE The current entity type to create
123 * @param {Integer} xPos The x position to create the object at
124 * @param {Integer} yPos The y position to create the object at
125 * @param {Image} SpriteSheet The currently loaded sprite sheet for the entity
126 * @param {AIController | Input} Controller The current entities controller Player or AI
127 * @return {Entity} Returns the Entity created
128 */
129 EngineEvent.prototype.CreateEntity = function(TYPE, xPos, yPos, SpriteSheet, Controller) {
130     var temp = new Entity(TYPE, xPos, yPos, SpriteSheet, Controller);
131     return(temp);
132 }
133 /**
134 * Function to display the event output(Simply output a text representation of everything in the object)
135 * Create and return an Entity object
136 */
137 EngineEvent.prototype.Display = function() {
138 	alert(this.ApplyFrom + " \n" + this.ApplyTo + " \n" + this.EventType + " \n" + this.InstanceAtX + " \n" + this.InstanceAtY + " \n" + this.ApplicationStop + " \n" + this.args);
139 }
