 17 /**
 18  * @class
 19  * Class modeling a player controller
 20  *
 21  * @description
 22  * A player is modeled as a controller for all intensive purposes.
 23  **/
 24 /**
 25 * @constructor
 26 */
 27 var Input = function(TVZ_ControlType) {
 28 	this.ControlType = TVZ_ControlType;
 29 	this.UP = null;
 30 	this.DOWN = null;
 31 	this.LEFT = null;
 32 	this.RIGHT = null;
 33 	this.SHOOT = null;
 34 	this.MENU = null;
 35 	this.MousePositionX = null;
 36 	this.MousePositionY = null;
 37 	this.CHWEAPON = null; //Change Weapon
 38     this.SetMappings();	
 39 }
 40 //---------------------------------------------------------------------MOUSE---------------------------------------------------
 41 /**
 42 * Returns the Mouse position 
 43 * @return {Array[]} Returns the current 2D mouse position 
 44 */
 45 Input.prototype.GetMousePosition = function(canvas, e){
 46 	if(this.ControlType == "Mouse"){
 47 		return(new Array(this.MousePositionX, this.MousePositionY));
 48 	}
 49 }
 50 /**
 51 * Sets the current Mouse position
 52 */
 53 Input.prototype.SetMousePosition = function(PosX, PosY){
 54 	if(PosX != null && PosY != null){
 55 		this.MousePositionX = PosX;
 56 		this.MousePositionY = PosY;
 57 	}
 58 }
 59 //---------------------------------------------------------------------GET ACCESSORS-------------------------------------------
 60 /**
 61 * Returns the current control type
 62 * @return {String} Returns the current control type string
 63 */
 64 Input.prototype.GetControlType = function(){
 65     return(this.ControlType);
 66 }
 67 /**
 68 * Event handler for key presses
 69 * @return {Array[]} Returns the current Input object
 70 */
 71 Input.prototype.GetMappings = function() {
 72     var Map = [this.ControlType, this.UP, this.DOWN, this.LEFT, this.RIGHT, this.SHOOT, this.CHWEAPON];
 73     return(Map);
 74 }
 75 //---------------------------------------------------------------------SET ACCESSORS---------------------------------------------
 76 //---------------------------------------------------------------------UTILITY FUNCTIONS-----------------------------------------
 77 /**
 78 * Sets the key mappings based upon provided type schema
 79 */
 80 Input.prototype.SetMappings = function(){
 81     switch(this.ControlType){
 82 		case "Mouse":
 83 			this.Mouse();
 84 		break;
 85         case "WASD":
 86 			this.WASD();
 87         break;
 88 		case "Arrow":
 89 			this.Arrow();
 90         break;
 91 		case "IJKL":
 92 			this.IJKL();
 93 		break;
 94         default:
 95 			this.WASD();
 96         break;
 97     }
 98 }
 99 //--------------------------------------------------------------------MOUSE CONTROL DEFINITIONS----------------------------------------

101 * Mouse Key Mappings

104 	this.LEFT = 65;
105 	this.RIGHT = 68;
106 	this.UP = 87;
107 	this.DOWN = 83;
108 	this.CHWEAPON = 67; //C

112 }
