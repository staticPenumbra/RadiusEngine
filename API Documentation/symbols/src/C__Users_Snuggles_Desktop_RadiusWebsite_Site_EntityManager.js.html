123 //Helper function for collision algorithm takes two point arrays
124 /**
125 * @param {Array[]} Object1 A reference to the colliding array of box points UL,UR,LL, LR
126 * @param {Array[]} Object2 A reference to the stationary array of box points UL,UR,LL, LR
127 */
128 EntityManager.prototype.CollisionCheck = function(Object1, Object2){
129 		//vert
130 		var cond1 = (Object1[0][1] > Object2[2][1]);
131 		//vert
132 		var cond2 = (Object1[2][1] < Object2[0][1]);
133 		//horiz
134 		var cond3 = (Object1[1][0] < Object2[0][0]);
135 		//horiz
136 		var cond4 = (Object1[0][0] > Object2[1][0]);
137 		if(!((cond1) || (cond2) || (cond3) || (cond4))){
138 		//We're not outside the box so we must be inside
139 		return(true);
140 		}
141 }
142 //Checks to make sure an entity hasn't tripped a trigger returns an array of fired triggers
143 /**
144 * @param {Trigger[]} Triggers A reference to an array of triggers to test
145 * @return {Trigger[]} A reference to an array of all the fired triggers in this application cycle
146 */
147 EntityManager.prototype.CheckTriggers = function(Triggers){
148 	if(Triggers != null){
149 		var FiredTriggers = new Array();
150 		for(var i = 0; i < Triggers.length; i++){
151 			var Trigger = Triggers[i];
152 			var TriggerPosition = Trigger.GetPosition();
153 			var TriggerCorners = Trigger.GetCorners();
154 			//For each Trigger try the list of application entities make sure to only process players
155 				for(var j = 0; j < this.Entities.length; j++){
156 					var Entity = this.Entities[j];
157 					if(Entity.GetReferenceName() == "Player"){
158 						var EntityPosition = Entity.GetPosition();
159 						var EntityVector = Entity.GetVelocity();
160 						var EntityCorners = Entity.GetCorners();
161 						//Compare the current position to see if the Entity is on the Trigger
162 						if(this.CollisionCheck(EntityCorners, TriggerCorners)){
163 							//We're not outside the box so we must be inside fire the trigger
164 							FiredTriggers.push(Trigger.Fire(Entity));
165 							//Make sure to remove the trigger from the processing list if it's fire once
166 							if(Trigger.GetFireOnce() == true){
167 								Triggers.splice(i, 1);
168 							}
169 						}
170 					}
171 				}
172 		}
173 		return(FiredTriggers);
174 	}
175 }

399 //Function to update all of the Entity positions every cycle
400 EntityManager.prototype.UpdateEntities = function(){
401     if(this.Entities != null && this.Entities.length > 0){
402     //We have entities to update
403         for(var i = 0; i < this.Entities.length; i++){
404             //Make sure the entity is on screen 
405             var Removed = this.CheckBounds(this.Entities[i]);
406 			//Need to decrement i after an element is deleted and the number of entities is reduced
407 			if(Removed > 0){
408 				i = i - Removed;
409 			}
410             //Update Entity Position
411             Removed = this.UpdatePosition(this.Entities[i]);
412 			//we need to decrement i if any removed
413 			if(Removed > 0){
414 				i = i - Removed;
415 			}
416             //If the Entity is an AI
417 			if(!(this.Entities[i])){
418 			 alert("Undefined Entity");
419 			}
420                 if(this.Entities[i].GetType() == "zombie"){
421                     this.AIManager.SetCurrentEntity(this.Entities[i]);
422                     this.AIManager.RunRoutine(this.Entities[i]);
423                 }
424                 if(this.Entities[i].GetType() == "TVZ_pistolBullet"){
425                     this.AIManager.SetCurrentEntity(this.Entities[i]);
426                     this.AIManager.RunRoutine(this.Entities[i]);
427                 }
428                 if(this.Entities[i].GetType() == "TVZ_shotgunBullet"){
429                     this.AIManager.SetCurrentEntity(this.Entities[i]);
430                     this.AIManager.RunRoutine(this.Entities[i]);
431                 }
432                 if(this.Entities[i].GetType() == "TVZ_rifleBullet"){
433                     this.AIManager.SetCurrentEntity(this.Entities[i]);
434                     this.AIManager.RunRoutine(this.Entities[i]);
435                 }
436         }
437         this.AIManager.UpdateEntities(this.Entities);    
438     }
439 }
440 //Returns an array of 4 coordinate points UpperL, UpperR, LowerL, LowerR
441 /**
442 * @param {Entity} Entity1 The Entity to process
443 * @returns {Array[]} Returns the array of x,y points [Upper Left, Upper Right, Lower Left, Lower Right]
444 */
445 EntityManager.prototype.GetCornerPoints = function(Entity){
446     if(Entity != null){
447         var ImageDimensions = Entity.GetImageDimensions();
448         //Upper Left
449         var UL = Entity.GetPosition();
450         //Upper Right
451         var UR = new Array(UL[0] + ImageDimensions[0], UL[1]);
452         //Lower Left
453         var LL = new Array(UL[0], UL[1] + ImageDimensions[1]);
454         //Lower Right
455         var LR = new Array(UL[0] + ImageDimensions[0], UL[1] + ImageDimensions[1]);
456         return(new Array(UL, UR, LL, LR));
457     }

459 //Applies Damage and returns the number of killed entities
460 /**
461 * @param {Entity} Weapon Colliding/Damaging Entity
462 * @param {Entity} Victim The Victim taking damage
463 * @return {Integer} Integer representing the number of killed Entities
464 */
465 EntityManager.prototype.DoDamage = function(Weapon, Victim){
466 	if(Weapon != null && Victim != null){
467 		var killed = 0;
468 		switch(Weapon.GetType()){
469 		    //Pistol Bullet Collision Handler
470 			case "TVZ_pistolBullet":
471 				var dead = Victim.TakeDamage(Weapon.GetDamage());
472 				if(dead == 0){
473 					this.RemoveEntity(Victim);
474 					killed++;
475 				}
476 				this.RemoveEntity(Weapon);
477 				killed++;
478 			break;
479 			case "TVZ_shotgunBullet":
480 				var dead = Victim.TakeDamage(Weapon.GetDamage());
481 				if(dead == 0){
482 					this.RemoveEntity(Victim);
483 					killed++;
484 				}
485 				this.RemoveEntity(Weapon);
486 				killed++;
487 			break;
488 			case "TVZ_rifleBullet":
489 				var dead = Victim.TakeDamage(Weapon.GetDamage());
490 				if(dead == 0){
491 					this.RemoveEntity(Victim);
492 					killed++;
493 				}
494 				this.RemoveEntity(Weapon);
495 				killed++;
496 			break;
497 			//Zombie Collision Handler
498 			case "zombie":
499 				if(Victim.GetType() != "zombie"){
500 					var dead = Victim.TakeDamage(Weapon.GetDamage());
501 					if(dead == 0){
502 						this.RemoveEntity(Victim);
503 						killed++;
504 					}
505 				}
506 			break;
507 			default:
508 			break;
509 		}
510 		return(killed);
511 	}
512 }
513 //Function to fix the orientation of 2 colliding entities
514 /**
515 * @param {Entity} Mover The Moving Entity
516 * @param {Entity} Obstacle The Entity to avoid collison with
517 * @param {String} MoverLocation The Direction the Mover is from the obstacle.  Valid: n, ne, e, se, s, sw, w, nw 
518 */
519 EntityManager.prototype.SetSnug = function(Mover, Obstacle, MoverLocation){
520     var MoverVector = Mover.GetVelocity();
521     var MoverCorners = Mover.GetCorners();//[Upper Left, Upper Right, Lower Left, Lower Right]
522     var ObstacleCorners = Obstacle.GetCorners();//[Upper Left, Upper Right, Lower Left, Lower Right]
523     switch(MoverLocation){
524     case 'n':
525     if(MoverVector[0] == 's'){
526         if(MoverVector[1] + MoverCorners[2][1] > ObstacleCorners[0][1]){
527             Mover.ApplyVelocity('s', (ObstacleCorners[0][1] - 1) - MoverCorners[2][1]);
528         }
529     }
530     break;
531     case 's':
532     if(MoverVector[0] == 'n'){
533         if(MoverCorners[0][1] - MoverVector[1] < ObstacleCorners[2][1]){
534             Mover.ApplyVelocity('n', MoverCorners[0][1] - ObstacleCorners[2][1] + 1);
535         }
536     }
537     break;
538     case 'e':
539     if(MoverVector[0] == 'w'){
540         if(MoverCorners[0][0] - MoverVector[1] < ObstacleCorners[1][0]){
541             Mover.ApplyVelocity('w', MoverCorners[0][0] - (ObstacleCorners[1][0] + 1));
542         }
543     }
544     break;
545     case 'w':
546     if(MoverVector[0] == 'e'){
547         if(MoverCorners[1][0] + MoverVector[1] > ObstacleCorners[0][0]){
548             Mover.ApplyVelocity('e', (ObstacleCorners[0][0] - 1) - MoverCorners[1][0]);
549         }
550     }
551     break;
552     default:
553 		alert("EntityManager SetSnug Unknown Direction");
554     break;
555     }                            
556 }
557 //Function to stop an entity
558 /**
559 * @param {Entity} Entity The Entity to stop 
560 */
561 EntityManager.prototype.StopEntity = function(Entity){
562 	var Vector = Entity.GetVelocity();
563 	var Pos = Entity.GetPosition();
564 	switch(Vector[0]){
565 		case "n":
566 			Entity.ApplyVelocity("c", 0);
567 			Entity.Move(Pos[0], Pos[1] + Vector[1]);
568 		break;
569 		case "s":
570 			Entity.ApplyVelocity("c", 0);
571 			Entity.Move(Pos[0], Pos[1] - Vector[1]);
572 		break;
573 		case "e":
574 			Entity.ApplyVelocity("c", 0);
575 			Entity.Move(Pos[0] - Vector[1], Pos[1]);
576 		break;
577 		case "w":
578 			Entity.ApplyVelocity("c", 0);
579 			Entity.Move(Pos[0] + Vector[1], Pos[1]);
580 		break;
581 		case "ne":
582 			Entity.ApplyVelocity("c", 0);
583 			Entity.Move(Pos[0] - Vector[1], Pos[1] + Vector[1]);
584 		break;
585 		case "se":
586 			Entity.ApplyVelocity("c", 0);
587 			Entity.Move(Pos[0] - Vector[1], Pos[1] - Vector[1]);
588 		break;
589 		case "nw":
590 			Entity.ApplyVelocity("c", 0);
591 			Entity.Move(Pos[0] + Vector[1], Pos[1] + Vector[1]);
592 		break;
593 		case "sw":
594 			Entity.ApplyVelocity("c", 0);
595 			Entity.Move(Pos[0] + Vector[1], Pos[1] - Vector[1]);
596 		break;
597 		default:
598 			alert("Unknown wall contact direction");
599 		break;
600 	}	
601 }
602 //Function to check an entity for collision against all other application entities
603 /**
604 * @param {Entity} Entity The Entity to check 
605 * @returns {Integer} The number of killed entities
606 */
607 EntityManager.prototype.CheckAllCollisions = function(Entity){
608 	//Make sure we are passed valid Input
609 	if(Entity != null){
610 		var E1Corners = Entity.GetCorners();

716 //Function to check if an entity is within screen bounds and stop it if it is going over
717 /**
718 * @param {Entity} Entity The Entity to check bounds on
719 * @returns {Integer} The number of killed objects
720 */
721 EntityManager.prototype.CheckBounds = function(Entity){
722     //Make sure there is an Entity being passed
723     if(Entity != null){
724         var Pos = Entity.GetPosition();
725         var Velo = Entity.GetVelocity();
726         var pts = this.GetCornerPoints(Entity);
727 		var killed = 0;
728         if(Velo[0] == 'w' && Pos[0] - Velo[1] <= 0){ 
729             //Reset Vector
730             Entity.ApplyVelocity('w', Pos[0] - 1);
731             killed = this.KillBullets(Entity);
732         }
733         if(Velo[0] == 'e' && pts[1][0] + Velo[1] >= this.ResolutionX - 1){
734             //Reset Vector
735             Entity.ApplyVelocity('e', (this.ResolutionX - 1) - pts[1][0]);
736             killed = this.KillBullets(Entity);
737         }
738         if(Velo[0] == 'n' && Pos[1] - Velo[1] <= 0){
739             //Reset Vector
740             Entity.ApplyVelocity('n', Pos[1] - 1);
741             killed = this.KillBullets(Entity);
742         } 
743         if(Velo[0] == 's' && pts[2][1] + Velo[1] >= this.ResolutionY - 1){
744             //Reset Vector
745             Entity.ApplyVelocity('s', (this.ResolutionY - 1) - pts[2][1]);
746             killed = this.KillBullets(Entity);
747         }  
748 		return(killed);
749     }

768             removed = this.CheckAllCollisions(Entity);
769             break;
770             case 'w':

776             removed = this.CheckAllCollisions(Entity);
777             break;
778 			case 'ne':
779 			Entity.Move(Entity.GetPosition()[0] + Entity.GetVelocity()[1], Entity.GetPosition()[1] - Entity.GetVelocity()[1]);
780             removed = this.CheckAllCollisions(Entity);
781 			break;
782 			case 'se':

784             removed = this.CheckAllCollisions(Entity);
785 			break;

789 			break;
